Example
// standard
[].xAdd('a'); // ['a']
[].xAdd( 'a', 'b' ); // ['a','b']
[].xAdd([ 'a', 'b' ]); // ['a','b']
[].xAdd(['a'], ['b']); // ['a','b']

// empty items
[].xAdd(); // []
[].xAdd([]); // []
[].xAdd(['a'], []); // ['a']

// undefined items
[].xAdd( undefined ); // []
[].xAdd( undefined, 'b' ); // ['b']
[].xAdd([ 'a', undefined ]); // ['a']
[].xAdd(['a'], [ undefined ]); // ['a']

Examples
['A','B'].xEach({
  success: function( item, index ) {
    // return; go to the next item 
    // throw new Break(); stop each of items
    console.log('success:', index, item );
  },
  complete: function() {
    console.log('complete');
  }
});
console.log('end');
// logs:
