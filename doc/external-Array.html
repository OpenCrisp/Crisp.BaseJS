xAdd(item) → {this}

add one or more items/arrays for concat in Array. empty Arrays and undefined items are ignored
Parameters:
Name Type Description
item external:Object | external:Array one or more of items
Source:
  • global-Array.js, line 14
Returns:
Type
this
Example
// standard
[].xAdd('a'); // ['a']
[].xAdd( 'a', 'b' ); // ['a','b']
[].xAdd([ 'a', 'b' ]); // ['a','b']
[].xAdd(['a'], ['b']); // ['a','b']

// empty items
[].xAdd(); // []
[].xAdd([]); // []
[].xAdd(['a'], []); // ['a']

// undefined items
[].xAdd( undefined ); // []
[].xAdd( undefined, 'b' ); // ['b']
[].xAdd([ 'a', undefined ]); // ['a']
[].xAdd(['a'], [ undefined ]); // ['a']

Examples
['A','B'].xEach({
  success: function( item, index ) {
    // return; got to the next item 
    // throw new Break(); stop each of items
    console.log('success:', index, item );
  },
  complete: function() {
    console.log('complete');
  }
});
console.log('end');
// logs:
// success: 0 A
// success: 1 B
// complete
// end
// async
['A','B'].xEach({
  async: true,
  success: function( item, index ) {
    console.log('success:', index, item );
  },

    console.log('complete');
  }
});
console.log('end');
// logs:
// end
// success: 0 A

// complete
